18.1 開場故事情境
林政宏是台北市交通局的資料分析師。每個月月底，他都要花兩到三天的時間處理捷運運量報告：從各站下載 CSV 原始資料、用 Excel 手動整理表格、複製貼上到 PowerPoint、再逐一撰寫分析文字。這個流程讓他每次都精疲力竭，而且每換一個月的資料，所有步驟都要重來一遍。

18.2 核心概念：資料視覺化 + AI 解讀的協同架構
為什麼需要 AI 解讀資料？
傳統資料視覺化工具（Excel、Tableau、Power BI）只負責「畫圖」。圖畫出來之後，人要自己「看懂」，再自己「寫結論」。

AI 解讀資料改變了這個流程：

圖表：讓人類快速掌握整體形狀和趨勢
AI 洞察：自動發現異常點、計算同比增減、提出可能的原因假設
協同效果：人類看圖 + AI 解讀文字 = 比任何一種單獨使用更強的分析效果

方法 C：前端摘要後傳遞（推薦，最彈性）

用 JavaScript 先計算統計摘要（最大值、最小值、月增率），再把摘要傳給 Gemini，兼顧速度與深度。

// 前端計算統計摘要
function buildSummary(data) {
  const totalsByStation = {};
  data.forEach(row => {
    const station = row['站名'];
    const count = parseInt(row['進站人次'] || 0);
    totalsByStation[station] = (totalsByStation[station] || 0) + count;
  });
  
  const sorted = Object.entries(totalsByStation)
    .sort((a, b) => b[1] - a[1]);
  
  return {
    top5: sorted.slice(0, 5),
    bottom5: sorted.slice(-5),
    total: sorted.reduce((s, [, v]) => s + v, 0)
  };
}

function buildDataSummary(data) {
  const stationMap = {};
  const monthMap = {};
  let totalEntries = 0;
  
  data.forEach(row => {
    const station = row['站名'] || row['station'] || '';
    const date = row['日期'] || row['date'] || '';
    const entries = parseInt(row['進站人次'] || row['entries'] || 0);
    const month = date.substring(0, 7); // YYYY-MM
    
    if (!station) return;
    
    // 站別累計
    stationMap[station] = (stationMap[station] || 0) + entries;
    
    // 月別累計
    if (!monthMap[month]) monthMap[month] = {};
    monthMap[month][station] = (monthMap[month][station] || 0) + entries;
    
    totalEntries += entries;
  });
  
  // 排序取前 10
  const topStations = Object.entries(stationMap)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10);
  
  const months = Object.keys(monthMap).sort();
  
  return { stationMap, monthMap, topStations, months, totalEntries };
}

async function analyzeWithGemini(summary) {
  const API_KEY = 'YOUR_GEMINI_API_KEY'; // 替換成你的金鑰
  
  // 建立傳給 AI 的摘要文字
  const summaryText = `
台北捷運運量資料摘要：
- 統計月份：${summary.months.join('、')}
- 總進站人次：${(summary.totalEntries / 10000).toFixed(0)} 萬人次
- 統計車站數：${Object.keys(summary.stationMap).length} 站

前十名車站進站人次：
${summary.topStations.map(([s, v], i) => 
  `${i+1}. ${s}：${(v/10000).toFixed(0)} 萬人次`
).join('\n')}

月度趨勢（前五大站）：
${summary.months.map(m => {
  const top5 = summary.topStations.slice(0, 5);
  return `${m}：${top5.map(([s]) => 
    `${s} ${((summary.monthMap[m]||{})[s]||0)/10000|0}萬`
  ).join('、')}`;
}).join('\n')}
  `.trim();
  
  try {
    const response = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${API_KEY}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{
            parts: [{
              text: `你是台北捷運的資深數據分析師。以下是捷運運量的統計摘要，請提供專業的分析洞察：

${summaryText}

請用繁體中文回答，分析以下面向：
1. 整體運量趨勢（成長、下降或穩定）
2. 前三名熱門站的特性分析（為什麼這些站運量高？）
3. 有沒有異常的月份或站別？
4. 給交通局的具體建議（2-3 條）

回答格式要清晰有條理，每段有小標題。`
            }]
          }],
          generationConfig: {
            temperature: 0.7,
            maxOutputTokens: 1024
          }
        })
      }
    );
    
    const result = await response.json();
    const text = result.candidates?.[0]?.content?.parts?.[0]?.text || '分析失敗，請檢查 API 金鑰。';
    
    document.getElementById('insightText').textContent = text;
    document.getElementById('spinner').style.display = 'none';
    
  } catch (error) {
    document.getElementById('insightText').textContent = 
      `無法取得 AI 分析：${error.message}\n\n請確認 API 金鑰是否正確。`;
    document.getElementById('spinner').style.display = 'none';
  }
}

def load_mrt_csv(filepath: str) -> pd.DataFrame:
    """
    讀取台北捷運 CSV 資料
    預期欄位：日期, 站名, 進站人次, 出站人次
    """
    df = pd.read_csv(filepath, encoding='utf-8-sig')
    
    # 欄位名稱標準化
    col_map = {
        '日期': 'date',
        '站名': 'station',
        '進站人次': 'entries',
        '出站人次': 'exits'
    }
    df = df.rename(columns={k: v for k, v in col_map.items() if k in df.columns})
    
    # 轉換日期
    df['date'] = pd.to_datetime(df['date'], errors='coerce')
    df['month'] = df['date'].dt.to_period('M')
    df['entries'] = pd.to_numeric(df['entries'], errors='coerce').fillna(0)
    
    return df


def compute_summary(df: pd.DataFrame) -> dict:
    """計算統計摘要"""
    station_total = df.groupby('station')['entries'].sum().sort_values(ascending=False)
    monthly_total = df.groupby('month')['entries'].sum()
    top_stations = station_total.head(10)
    
    summary = {
        "total_entries": int(df['entries'].sum()),
        "station_count": df['station'].nunique(),
        "date_range": f"{df['date'].min().date()} ~ {df['date'].max().date()}",
        "top_stations": top_stations.to_dict(),
        "monthly_trend": {str(k): int(v) for k, v in monthly_total.items()},
        "top5_monthly": {}
    }
    
    top5 = station_total.head(5).index.tolist()
    for station in top5:
        station_df = df[df['station'] == station]
        monthly = station_df.groupby('month')['entries'].sum()
        summary["top5_monthly"][station] = {str(k): int(v) for k, v in monthly.items()}
    
    return summary

def analyze_with_gemini(summary: dict) -> str:
    """呼叫 Gemini 3 Flash 進行 AI 洞察分析"""
    
    model = genai.GenerativeModel("gemini-3-flash-preview")
    
    # 建立結構化摘要文字
    top_stations_text = "\n".join([
        f"  {i+1}. {station}：{count/10000:.1f} 萬人次"
        for i, (station, count) in enumerate(list(summary["top_stations"].items())[:10])
    ])
    
    monthly_text = "\n".join([
        f"  {month}：{count/10000:.1f} 萬人次"
        for month, count in summary["monthly_trend"].items()
    ])
    
    prompt = f"""你是台北捷運的資深數據分析師，擅長從運量數據中提取有意義的洞察。

以下是台北捷運的運量統計摘要：

【基本資訊】
- 統計期間：{summary['date_range']}
- 總進站人次：{summary['total_entries']/1000000:.1f} 百萬人次
- 統計車站數：{summary['station_count']} 站

【前十名熱門車站（進站總人次）】
{top_stations_text}

【月度整體趨勢】
{monthly_text}

請以專業分析師的角度，提供以下洞察分析（用繁體中文，條理清晰）：

## 整體趨勢分析
（說明整體運量是成長、下降還是穩定，以及可能的原因）

## 熱點站別解析
（分析前三名車站的高運量原因，結合地理位置和商業特性）

## 異常與警訊
（有無運量異常的月份或站別，可能是什麼原因）

## 政策建議
（給台北市交通局的具體改善建議，2-3 條）
"""
    
    response = model.generate_content(prompt)
    return response.text


def run_dashboard(csv_path: str, output_dir: str = "output"):
    """主程式：完整執行分析流程"""
    print("=" * 60)
    print("  台北捷運運量分析儀表板  ")
    print("=" * 60)
    
    # 步驟一：讀取資料
    print("\n[1/4] 讀取 CSV 資料...")
    df = load_mrt_csv(csv_path)
    print(f"  載入 {len(df):,} 筆資料，{df['station'].nunique()} 個站")
    
    # 步驟二：計算摘要
    print("\n[2/4] 計算統計摘要...")
    summary = compute_summary(df)
    print(f"  總進站：{summary['total_entries']/1000000:.1f} 百萬人次")
    print(f"  期間：{summary['date_range']}")
    
    # 步驟三：生成圖表
    print("\n[3/4] 生成視覺化圖表...")
    chart_path = create_visualizations(df, summary, output_dir)
    
    # 步驟四：AI 洞察分析
    print("\n[4/4] Gemini AI 洞察分析中...")
    insight = analyze_with_gemini(summary)
    
    # 輸出報告
    report_path = Path(output_dir) / "mrt_insight_report.txt"
    with open(report_path, 'w', encoding='utf-8') as f:
        f.write("台北捷運運量 AI 洞察報告\n")
        f.write("=" * 60 + "\n\n")
        f.write(insight)
    
    print(f"\n  AI 洞察報告已儲存至：{report_path}")
    print("\n--- AI 洞察摘要 ---")
    print(insight[:500] + "..." if len(insight) > 500 else insight)
    
    print("\n分析完成！")
    return summary, insight


if __name__ == "__main__":
    # 執行示範
    run_dashboard(
        csv_path="mrt_data.csv",   # 你的 CSV 路徑
        output_dir="output"
    )
